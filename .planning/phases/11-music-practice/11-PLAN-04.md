Props interface:
  interface PitchSample { note: string; midi: number; cents: number; ts: number }

Imports: useRef, useEffect from 'react'; Panel from '../../panels/Panel'.

Constants (define at module level, not inside component):
  MIDI_MIN = 36  (C2)
  MIDI_MAX = 84  (C6)
  VISIBLE_SEMITONES = MIDI_MAX - MIDI_MIN + 1  (= 49)
  C_NOTES = [36, 48, 60, 72, 84]  (MIDI values of C2–C6 for labels)
  NOTE_NAMES_SHARP = ['C','C#','D','D#','E','F','F#','G','G#','A','A#','B']

Helper functions at module level:
  pitchColor(cents: number): string
    if Math.abs(cents) <= 10: return 'rgba(0,255,255,0.9)'   // cyan
    if Math.abs(cents) <= 25: return 'rgba(255,176,0,0.9)'   // amber
    return 'rgba(255,0,128,0.9)'                               // magenta

  midiToRow(midi: number, canvasHeight: number): number
    // Y position for a given MIDI note. MIDI_MAX at top (y=0), MIDI_MIN at bottom.
    const rowH = canvasHeight / VISIBLE_SEMITONES
    return (MIDI_MAX - midi) * rowH

Component PitchRollPanel({ pitchBufRef }):
  const canvasRef = useRef<HTMLCanvasElement>(null)
  const rafRef = useRef<number>(0)
  const offscreenRef = useRef<HTMLCanvasElement | null>(null)

  Helper drawGrid(canvas: HTMLCanvasElement, offscreen: HTMLCanvasElement):
    // Draw the static chromatic grid onto the offscreen canvas.
    // Reads CSS tokens via getComputedStyle(document.documentElement) for color vars.
    // Background: var(--bg-elevated) for white keys, var(--bg-base) for black keys.
    // Black key MIDI rows: C#, D#, F#, G#, A# (midi % 12 in [1,3,6,8,10])
    // Semitone grid lines: 1px border-color lines between rows
    // C-note labels: right-aligned text "C2", "C3" etc at each C row
    // Octave separator: slightly brighter line at each C note
    // Grid is drawn once; redrawn only on resize.

  Helper drawFrame(canvas: HTMLCanvasElement):
    const ctx = canvas.getContext('2d')!
    const { width, height } = canvas
    const dpr = window.devicePixelRatio || 1
    const cssW = width / dpr
    const cssH = height / dpr

    // 1. Draw static grid from offscreen
    ctx.drawImage(offscreenRef.current!, 0, 0)

    // 2. Scroll existing pitch data left by 1px (using drawImage self-shift)
    //    Allocate a single-pixel right column for new data.
    //    Use: ctx.drawImage(canvas, -1, 0) — shifts entire canvas content 1px left.
    //    This is the efficient pixel-scroll technique for Canvas 2D.
    //    Then overdraw the grid for the leftmost column that would become dirty:
    //    Actually: draw grid first, then self-shift the pitch trail separately.
    //    Pattern: use two layers — grid (offscreen, static) + trail (self-shift).
    //    Correct implementation:
    //      a. ctx.drawImage(offscreenRef.current!, 0, 0)   // grid underneath
    //      b. If there is stored trail data: apply self-shift to the trail portion
    //    Simpler correct approach: maintain a separate offscreen "trail" canvas,
    //    shift it left by 1 each frame, draw new column, then composite:
    //      ctx.drawImage(offscreenRef.current!, 0, 0)      // grid
    //      ctx.drawImage(trailCanvas, 0, 0)                // trail on top
    //    Use this two-offscreen approach.

    // 3. Paint rightmost column with latest pitch sample (if any)
    const latest = pitchBufRef.current[pitchBufRef.current.length - 1]
    if (latest && latest.midi >= MIDI_MIN && latest.midi <= MIDI_MAX) {
      const color = pitchColor(latest.cents)
      const rowH = cssH / VISIBLE_SEMITONES
      const y = midiToRow(latest.midi, cssH)
      // Glow effect
      ctx.shadowColor = color
      ctx.shadowBlur = 6
      ctx.fillStyle = color
      ctx.fillRect(cssW - 2, y, 2, rowH)
      ctx.shadowBlur = 0
    }

  rAF loop useEffect (empty deps []):
    1. Create offscreen grid canvas (offscreenRef.current = document.createElement('canvas'))
    2. Set up ResizeObserver on canvasRef to update canvas dimensions + DPR scaling + redraw grid:
       const dpr = window.devicePixelRatio || 1
       canvas.width = entry.contentRect.width * dpr
       canvas.height = entry.contentRect.height * dpr
       canvas.style.width = entry.contentRect.width + 'px'
       canvas.style.height = entry.contentRect.height + 'px'
       ctx.scale(dpr, dpr)
       drawGrid(canvas, offscreenRef.current)
    3. MutationObserver on document.documentElement for data-theme attribute changes:
       on change: redraw grid (reads fresh CSS tokens)
    4. Start rAF: function frame() { drawFrame(canvasRef.current!); rafRef.current = requestAnimationFrame(frame) }
       rafRef.current = requestAnimationFrame(frame)
    5. Cleanup: cancelAnimationFrame(rafRef.current); resizeObs.disconnect(); mutObs.disconnect()

Render:
  return (
    <Panel id="pitch-roll" title="Pitch Roll" noPad>
      <canvas ref={canvasRef} style={{ width: '100%', height: '100%', display: 'block' }} />
    </Panel>
  )
